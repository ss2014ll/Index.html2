    function loadQuiz() {
        const quizContainer = document.getElementById('quiz');
        let output = '';
        questions.forEach((q, qIdx) => {
            output += `<div class="question" id="qN-${qIdx}">
                <div class="question-title">${qIdx + 1}. ${q.q}</div>
                <div class="options">`;
            q.options.forEach((opt, oIdx) => {
                output += `<label class="option">
                    <input type="radio" name="question-${qIdx}" value="${oIdx}"> <span>${opt}</span>
                </label>`;
            });
            output += `</div><div class="feedback" id="fb-${qIdx}">💡 <b>مرجع تكتيكي:</b> ${q.hint}</div></div>`;
        });
        quizContainer.innerHTML = output;
    }

    function checkAnswers() {
        let score = 0;
        questions.forEach((q, qIdx) => {
            const selected = document.querySelector(`input[name="question-${qIdx}"]:checked`);
            const questionElement = document.getElementById(`qN-${qIdx}`);
            const feedbackElement = document.getElementById(`fb-${qIdx}`);
            
            feedbackElement.style.display = 'block';
            
            if (selected) {
                const answerIdx = parseInt(selected.value);
                if (answerIdx === q.answer) {
                    score++;
                    questionElement.classList.add('correct');
                    questionElement.classList.remove('incorrect');
                } else {
                    questionElement.classList.add('incorrect');
                    questionElement.classList.remove('correct');
                }
            } else {
                questionElement.classList.add('incorrect');
            }
        });

        const resultDiv = document.getElementById('result');
        resultDiv.style.display = 'block';
        const percent = Math.round((score / questions.length) * 100);
        resultDiv.innerHTML = `النتيجة النهائية الكلية: ${score} من ${questions.length} أسئلة صحيحة (${percent}%)`;
        window.scrollTo(0, 0);
    }

    loadQuiz();
